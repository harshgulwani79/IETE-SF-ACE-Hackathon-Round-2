        function switchTab(tab) {
            const learningSection = document.getElementById('learning-section');
            const managementSection = document.getElementById('management-section');
            const [learningTab, managementTab] = document.querySelectorAll('header button');
            
            if (tab === 'learning') {
                learningSection.classList.remove('hidden');
                managementSection.classList.add('hidden');
                learningTab.classList.add('tab-active');
                managementTab.classList.remove('tab-active');
            } else {
                learningSection.classList.add('hidden');
                managementSection.classList.remove('hidden');
                learningTab.classList.remove('tab-active');
                managementTab.classList.add('tab-active');
